package com.shaded.fasterxml.jackson.databind.util;

public final class LinkedNode {
    final LinkedNode _next;
    final T _value;

    public LinkedNode(T t, LinkedNode linkedNode) {
        this._value = t;
        this._next = linkedNode;
    }

    public LinkedNode next() {
        return this._next;
    }

    public T value() {
        return this._value;
    }

    public static  boolean contains(LinkedNode linkedNode, ST st) {
        while (linkedNode != null) {
            if (linkedNode.value() == st) {
                return true;
            }
            linkedNode = linkedNode.next();
        }
        return false;
    }
}
