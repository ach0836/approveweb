package com.firebase.client.collection;

import java.util.Comparator;

public interface LLRBNode {

    public enum Color {
        RED,
        BLACK
    }

    public interface ShortCircuitingNodeVisitor {
        boolean shouldContinue(K k, V v);
    }

    LLRBNode copy(K k, V v, Color color, LLRBNode lLRBNode, LLRBNode lLRBNode2);

    int count();

    K getKey();

    LLRBNode getLeft();

    LLRBNode getMax();

    LLRBNode getMin();

    LLRBNode getRight();

    V getValue();

    void inOrderTraversal(NodeVisitor nodeVisitor);

    LLRBNode insert(K k, V v, Comparator comparator);

    boolean isEmpty();

    boolean isRed();

    LLRBNode remove(K k, Comparator comparator);

    boolean shortCircuitingInOrderTraversal(ShortCircuitingNodeVisitor shortCircuitingNodeVisitor);

    boolean shortCircuitingReverseOrderTraversal(ShortCircuitingNodeVisitor shortCircuitingNodeVisitor);

    public static abstract class NodeVisitor implements ShortCircuitingNodeVisitor {
        public abstract void visitEntry(K k, V v);

        public boolean shouldContinue(K key, V value) {
            visitEntry(key, value);
            return true;
        }
    }
}
