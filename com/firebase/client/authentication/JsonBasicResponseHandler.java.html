package com.firebase.client.authentication;

import com.firebase.client.utilities.encoding.JsonHelpers;
import com.shaded.fasterxml.jackson.core.type.TypeReference;
import java.io.IOException;
import java.io.InputStream;
import java.util.Map;
import org.shaded.apache.http.HttpEntity;
import org.shaded.apache.http.HttpResponse;
import org.shaded.apache.http.client.ResponseHandler;

class JsonBasicResponseHandler implements ResponseHandler> {
    JsonBasicResponseHandler() {
    }

    public Map handleResponse(HttpResponse response) throws IOException {
        HttpEntity entity;
        Map map = null;
        if (!(response == null || (entity = response.getEntity()) == null)) {
            InputStream is = entity.getContent();
            try {
                map = (Map) JsonHelpers.getMapper().readValue(is, (TypeReference) new TypeReference>() {
                });
            } finally {
                is.close();
            }
        }
        return map;
    }
}
